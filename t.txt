package main

import (
	"fmt"
	"reflect"
)

type testStruct struct{}

func (tS *testStruct) A() {
	fmt.Println("Method OK!")
}
func launchMethod(nameMethod string) {
	st := &testStruct{}
	validMethod := reflect.ValueOf(st).MethodByName(nameMethod)
	if !validMethod.IsValid() {
		return
	}
	bar := func() {
		validMethod.Call(nil)
	}
	bar()
}
func main() {
	launchMethod("A")
}
